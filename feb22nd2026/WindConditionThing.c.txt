#include <stdio.h>
#include <stdbool.h>

int main(void)
{
	int speed = 0;

	printf("Enter your speed (knots): ");
	scanf_s("%d", &speed);

	if (speed < 1)
	{
		printf("Current wind speed: Calm");
	}
	else if (speed <= 3)
	{
		printf("Current wind speed: Light air");
	}
	else if (speed <= 27)
	{
		printf("Current wind speed: Breeze");
	}
	else if (speed <= 47)
	{
		printf("Current wind speed: Gale");
	}
	else if (speed <= 63)
	{
		printf("Current wind speed: Storm");
	}
	else
	{
		printf("Current wind speed: Hurricane!");
	}

	return 0;
}
