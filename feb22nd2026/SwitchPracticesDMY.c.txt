#include <stdio.h>
#include <stdbool.h>

int main(void)
{
	int day, month, year;

	printf("Enter your DD/MM/YY: ");
	scanf_s("%d/%d/%d", &day, &month, &year);


	// For this switch block specifically you can merge the cases into one by combining them on one line like: case 1: case 2: case 3: etc to simplify it
	// You then dont have to printf the entire statement in one case, you can just do: printf("st"); break; on the line below it for 1 for example
	switch (day)
	{
	case 1: printf("Dated this %dst", day);
		break;
	case 2: printf("Dated this %dnd ", day);
		break;
	case 3: printf("Dated this %drd ", day);
		break;
	case 21: printf("Dated this %dst ", day);
		break;
	case 22: printf("Dated this %dnd ", day);
		break;
	case 23: printf("Dated this %drd ", day);
		break;
	case 31: printf("Dated this %dst ", day);
		break;
	default: printf("Dated this %dth ", day);
		break;

	}


	// It would also make more sense to do printf("day of"); at the top here as then youd only need to store the month in the switch
	switch (month)
	{
	case 1: printf("day of January");
		break;
	case 2: printf("day of February");
		break;
	case 3: printf("day of March");
		break;
	case 4: printf("day of April");
		break;
	case 5: printf("day of May");
		break;
	case 6: printf("day of June");
		break;
	case 7: printf("day of July");
		break;
	case 8: printf("day of August");
		break;
	case 9: printf("day of September");
		break;
	case 10: printf("day of October");
		break;
	case 11: printf("day of November");
		break;
	case 12: printf("day of December");
		break;
	default: printf("day of Month");
		break;
	}

	printf(", 20%d", year);
	return 0;
}
